# PYTHON VERSION CONSTRAINT
# -------------------------
# Python >= 3.9 is REQUIRED (hard lower bound).
#   • `zoneinfo` (stdlib since 3.9) is used directly without a try/except
#     fallback in the pipeline source.  On Python 3.8 the package would fail
#     at import time with ModuleNotFoundError.
#   • `typing.Protocol` is stdlib since 3.8; no issue there.
#   • Tested on CPython 3.12.3.  Python 3.9–3.12 are the supported range.
#
# VERSION-PINNING STRATEGY
# ------------------------
# All packages use compatible-release bounds of the form:
#
#     package >= INSTALLED_VERSION, < NEXT_MAJOR
#
# Rationale:
#   • The lower bound (>=) guarantees the exact API surface that was tested.
#   • The upper bound (<NEXT_MAJOR) blocks silent breaking changes at the next
#     major version while allowing patch and security updates to flow through.
#   • This is the PEP 440 / PyPA recommended approach for production pinning
#     in a scientific replication context where reproducibility is paramount.
#   • An alternative is exact pinning (`==`); generate it at any time with:
#       pip freeze > requirements.lock
#

# ------------------------------------------------------------------------------
# Section 1 — Numerical / Data Science Core
# (Direct imports: pandas, numpy, scipy.stats)
# ------------------------------------------------------------------------------

# pandas — primary DataFrame engine for all tick, quote, metadata, and panel
# data manipulation throughout the pipeline.  Version 3.x introduces Copy-on-
# Write semantics (CoW); the pipeline is written to be CoW-compatible.
# Lower bound: 3.0.2 (tested).  Upper bound: <4 blocks the next breaking cycle.
pandas>=3.0.2,<4.0.0

# numpy — foundational N-dimensional array operations, including the quadratic
# covariation accumulator, covariance matrix construction, and log/exp
# transforms for the SDF entropy bound calculation.
# Upper bound <2.7 is imposed by scipy 1.17.x's own declared constraint
# (scipy requires numpy<2.7,>=1.26.4); exceeding it would break scipy.
numpy>=2.4.4,<2.7

# scipy — statistical distributions (scipy.stats) used for Cauchy-Schwarz
# bounding, winsorisation percentile computation, and the Alvarez-Jermann
# entropy integrals.
scipy>=1.17.1,<2.0.0


# ------------------------------------------------------------------------------
# Section 2 — Econometrics
# (Direct imports: statsmodels.api, linearmodels.panel.PanelOLS)
# ------------------------------------------------------------------------------

# statsmodels — OLS and GLS panel infrastructure (sm.OLS, sm.add_constant),
# heteroskedasticity-robust standard errors, and auxiliary diagnostics used
# in the cross-sectional factor regressions (Tasks 18–19).
statsmodels>=0.14.6,<1.0.0

# linearmodels — industry-standard Python package for high-dimensional fixed-
# effects panel regressions with multi-way clustered standard errors.
# PanelOLS is the primary estimator for the two-way (date × stock) FE
# regressions documented in the paper's Table panels.
# The pipeline raises ImportError explicitly if this package is absent.
linearmodels>=7.0,<8.0


# ------------------------------------------------------------------------------
# Section 3 — Configuration Parsing

# ------------------------------------------------------------------------------

# Required at module load time; absence causes FileNotFoundError or

PyYAML>=6.0.3,<7.0.0


# ------------------------------------------------------------------------------
# Section 4 — Timezone Data
# (Runtime requirement for stdlib zoneinfo on non-Windows platforms)
# ------------------------------------------------------------------------------

# tzdata — ships the IANA timezone database as a Python package.  On Linux/
# macOS the OS-level zoneinfo database is usually present, but its version
# varies by distribution and may be stale.  Pinning tzdata ensures that
# "America/New_York" (used for TAQ session alignment in
# microstructure_parameters.session_timezone) resolves consistently across
# all deployment environments regardless of the host OS timezone installation.
# On Windows, tzdata is a hard runtime requirement for zoneinfo.
tzdata>=2026.2,<2027.0
